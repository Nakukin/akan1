// Module navigation
function showModule(moduleName) {
// Hide all modules
document.querySelectorAll('.module-content').forEach(module => {
module.classList.add('hidden');
});

// Show selected module
document.getElementById(moduleName + '-module').classList.remove('hidden');

// Update navigation
document.querySelectorAll('.nav-item').forEach(item => {
item.classList.remove('bg-blue-50', 'text-blue-700');
item.classList.add('text-gray-700', 'hover:bg-gray-50');
});

event.target.classList.remove('text-gray-700', 'hover:bg-gray-50');
event.target.classList.add('bg-blue-50', 'text-blue-700');
}

// Sidebar toggle
function toggleSidebar() {
const sidebar = document.getElementById('sidebar');
const overlay = document.getElementById('sidebarOverlay');

sidebar.classList.toggle('-translate-x-full');
overlay.classList.toggle('hidden');
}

// Update current time
function updateTime() {
const now = new Date();
const timeString = now.toLocaleTimeString('id-ID');
const dateString = now.toLocaleDateString('id-ID', {
weekday: 'long',
year: 'numeric',
month: 'long',
day: 'numeric'
});

const timeElement = document.getElementById('currentTime');
const dateElement = document.getElementById('currentDate');

if (timeElement) timeElement.textContent = timeString;
if (dateElement) dateElement.textContent = dateString;
}

// Rating buttons functionality
function initRatingButtons() {
document.querySelectorAll('.rating-btn').forEach(btn => {
btn.addEventListener('click', function() {
const rating = this.dataset.rating;
const parent = this.parentElement;

// Reset all buttons in this group
parent.querySelectorAll('.rating-btn').forEach(b => {
b.classList.remove('border-yellow-400', 'bg-yellow-50');
b.classList.add('border-gray-300');
});

// Highlight selected rating and all previous
for (let i = 1; i <= rating; i++) {
const btn = parent.querySelector(`[data-rating="${i}"]`);
btn.classList.remove('border-gray-300');
btn.classList.add('border-yellow-400', 'bg-yellow-50');
}
});
});
}

// Check-in/Check-out functionality
function initAttendance() {
const checkInBtn = document.getElementById('checkInBtn');
const checkOutBtn = document.getElementById('checkOutBtn');

if (checkInBtn) {
checkInBtn.addEventListener('click', function() {
// Simulate check-in
this.disabled = true;
this.innerHTML = '<i class="fas fa-check mr-2"></i>Sudah Check In';
this.classList.remove('bg-green-600', 'hover:bg-green-700');
this.classList.add('bg-gray-400');

if (checkOutBtn) {
checkOutBtn.disabled = false;
checkOutBtn.classList.remove('bg-gray-400');
checkOutBtn.classList.add('bg-red-600', 'hover:bg-red-700');
}
});
}

if (checkOutBtn) {
checkOutBtn.addEventListener('click', function() {
// Simulate check-out
this.disabled = true;
this.innerHTML = '<i class="fas fa-check mr-2"></i>Sudah Check Out';
this.classList.remove('bg-red-600', 'hover:bg-red-700');
this.classList.add('bg-gray-400');
});
}
}

// Initialize everything when DOM is loaded
document.addEventListener('DOMContentLoaded', function() {
// Initialize charts
setTimeout(initCharts, 100);

// Initialize rating buttons
initRatingButtons();

// Initialize attendance
initAttendance();

// Update time every second
updateTime();
setInterval(updateTime, 1000);

// Close sidebar when clicking nav items on mobile
document.querySelectorAll('.nav-item').forEach(item => {

if (window.innerWidth < 1024) {
toggleSidebar();
}
});
});
});
